  \Tree [.S 
           [.NP [.N United ] ]
           [.\node [color = red] {VP};

Dependency Parse¶

\documentclass{article}
\usepackage[graphics,tightpage,active]{preview}
\usepackage{tikz}
\PreviewEnvironment{tikzpicture}
\usetikzlibrary{matrix}
\begin{document}
\begin{tikzpicture}

  % Draw the sentence. 
  \matrix (sentence) [baseline=(X.base), 
                     matrix of nodes, 
                     nodes in empty cells,
                     column sep = 0.5cm, 
                     nodes={text centered, baseline=(current bounding box.west)}] 
                     {
                        & & & &  \\
                       United & flies & some & large & jet \\
                     };
     
  % Draw the arcs. 
  \tikzstyle{arc} = [<-, thick];
  \tikzstyle{leftarc} = [arc, bend left=45];
  \tikzstyle{rightarc} = [arc, bend right=45];

  \draw (sentence-1-1.north) edge[leftarc] (sentence-1-2.north);
  \draw (sentence-1-3.north) edge[leftarc] (sentence-1-5.north);
  \draw (sentence-1-4.north) edge[leftarc] (sentence-1-5.north);
  \draw (sentence-1-5.north) edge[rightarc] (sentence-1-2.north);
\end{tikzpicture}
\end{document}
